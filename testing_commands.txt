# Terminal 1 (inference service)

cd packages/inference
source .venv/bin/activate
uvicorn main:app --reload --port 8000





# Terminal 2 (build shared if changes were made)

pnpm --filter @uttt/shared build



# Terminal 3 (NestJS backend)

pnpm --filter @uttt/backend dev





# Terminal 4 (example curl commands)

# 1. Create a game
ID=$(curl -s -X POST http://localhost:3000/games | python3 -c "import sys,json;print(json.load(sys.stdin)['id'])")
echo "Game ID: $ID"

# 2. Fetch it (should be empty board, player 1)
curl -s http://localhost:3000/games/$ID | python3 -m json.tool

# 3. Human (X) plays center-of-center
curl -s -X POST http://localhost:3000/games/$ID/moves \
  -H "Content-Type: application/json" \
  -d '{"move": [4, 4]}' | python3 -m json.tool

# 4. AI (O) responds — sims=100 keeps it fast for testing
curl -s -X POST "http://localhost:3000/games/$ID/ai-move?sims=100" | python3 -m json.tool
